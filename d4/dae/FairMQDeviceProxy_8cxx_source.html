55 void FairMQDeviceProxy::bindRoutes(std::vector<OutputRoute> const& outputs, fair::mq::Device& device)
56 {
57  mRoutes.reserve(outputs.size());
58  size_t ri = 0;
59  std::unordered_map<std::string, ChannelIndex> channelNameToChannel;
60  for (auto& route : outputs) {
61  // If the channel is not yet registered, register it.
62  // If the channel is already registered, use the existing index.
63  auto channelPos = channelNameToChannel.find(route.channel);
64  ChannelIndex channelIndex;
65 
66  if (channelPos == channelNameToChannel.end()) {
67  channelIndex = ChannelIndex{(int)mChannels.size()};
68  mChannels.push_back(&device.fChannels.at(route.channel).at(0));
69  channelNameToChannel[route.channel] = channelIndex;
70  LOGP(debug, "Binding channel {} to channel index {}", route.channel, channelIndex.value);
71  } else {
72  LOGP(debug, "Using index {} for channel {}", channelPos->second.value, route.channel);
73  channelIndex = channelPos->second;
74  }
75  LOGP(debug, "Binding route {}@{}%{} to index {} and channelIndex {}", route.matcher, route.timeslice, route.maxTimeslices, ri, channelIndex.value);
76  mRoutes.emplace_back(RouteState{channelIndex, false});
77  ri++;
78  }
79  for (auto& route : mRoutes) {
80  assert(route.channel.value != -1);
81  assert(route.channel.value < mChannels.size());
82  }
83  LOGP(debug, "Total channels found {}, total routes {}", mChannels.size(), mRoutes.size());
84  assert(mRoutes.size() == outputs.size());
85 }
